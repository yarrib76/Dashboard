Adjunto consulta ejemplo "use samira;
-- Eliminar tablas temporales si ya existen
DROP TEMPORARY TABLE IF EXISTS temp_cantPedidos;
DROP TEMPORARY TABLE IF EXISTS temp_dias;
DROP TEMPORARY TABLE IF EXISTS temp_totalFacturado;
DROP TEMPORARY TABLE IF EXISTS temp_cantArticulos;

SET SESSION sql_mode = 'STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION';
SET @ultActuDesde = '2024-06-01';
SET @ultActuHasta = '2024-06-31';

-- Crear tablas temporales para almacenar resultados intermedios
CREATE TEMPORARY TABLE temp_cantPedidos AS
SELECT vendedora, COUNT(*) AS cantPedidos
FROM samira.controlpedidos
WHERE ultactualizacion > @ultActuDesde
  AND ultactualizacion < @ultActuHasta
  AND total > 1
  AND estado <> 2
  AND ordenWeb > 0
GROUP BY vendedora;

CREATE TEMPORARY TABLE temp_dias AS
SELECT vendedora, COUNT(*) AS Dias
FROM (
    SELECT vendedora, DAY(ultactualizacion) AS dia
    FROM samira.controlpedidos
    WHERE ultactualizacion > @ultActuDesde
      AND ultactualizacion < @ultActuHasta
      AND total > 1
      AND estado <> 2
      AND ordenWeb > 0
    GROUP BY vendedora, dia
) AS subquery
GROUP BY vendedora;

CREATE TEMPORARY TABLE temp_totalFacturado AS
SELECT vendedora, ROUND(SUM(total), 2) AS TotalFacturado
FROM samira.controlpedidos
WHERE ultactualizacion > @ultActuDesde
  AND ultactualizacion < @ultActuHasta
  AND total > 1
  AND estado <> 2
  AND ordenWeb > 0
GROUP BY vendedora;

CREATE TEMPORARY TABLE temp_cantArticulos AS
SELECT controlpedidos.vendedora, COUNT(pedidotmp.Cantidad) AS cantArticulos
FROM samira.controlpedidos
INNER JOIN samira.pedidotemp AS pedidotmp ON pedidotmp.nropedido = controlpedidos.nropedido
WHERE controlpedidos.ultactualizacion > @ultActuDesde
  AND controlpedidos.ultactualizacion < @ultActuHasta
  AND controlpedidos.total > 1
  AND controlpedidos.ordenWeb > 0
  AND controlpedidos.estado <> 2
GROUP BY controlpedidos.vendedora;

-- Consulta final usando tablas temporales
SELECT t1.vendedora AS ResulVendedora,
       t1.cantPedidos,
       t2.Dias,
       ROUND(t1.cantPedidos / t2.Dias, 1) AS Promedio,
       ROUND(t3.TotalFacturado / t1.cantPedidos, 1) AS PromedioFacturado,
       ROUND(t4.cantArticulos / t1.cantPedidos, 1) AS PromedioCantArticulos
FROM temp_cantPedidos AS t1
INNER JOIN temp_dias AS t2 ON t1.vendedora = t2.vendedora
INNER JOIN temp_totalFacturado AS t3 ON t1.vendedora = t3.vendedora
INNER JOIN temp_cantArticulos AS t4 ON t1.vendedora = t4.vendedora;
"
